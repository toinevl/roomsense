## Task 1: Create roomFinder.ts page skeleton with occupancy filter

**Objective:** New page file with room list filtered by occupancy < capacity.

**Files:**
- Create: `frontend/src/pages/roomFinder.ts`
- Reuse: `frontend/src/lib/api.ts`, `frontend/src/lib/mockDerivations.ts`

**Code:**
```typescript
// frontend/src/pages/roomFinder.ts
import { apiClient } from '../lib/api'
import { html } from 'lit-html'

export async function roomFinder() {
  const rooms = await apiClient.rooms()
  
  // Filter: occupancy < capacity (room is available)
  const available = rooms.filter(r => (r.occupancy ?? 0) < r.capacity)
  
  // Sort by building, then floor
  available.sort((a, b) => {
    if (a.building !== b.building) return a.building.localeCompare(b.building)
    return (a.floor ?? 0) - (b.floor ?? 0)
  })

  return html`
    <div class="room-finder">
      <h1>Find a Room</h1>
      <p class="subtitle">Green = available now</p>
      <div class="room-cards">
        ${available.map(room => html`
          <div class="room-card">
            <div class="room-name">${room.name}</div>
            <div class="room-meta">${room.building} / Floor ${room.floor ?? 'N/A'}</div>
            <div class="occupancy">
              ${room.occupancy ?? 0} / ${room.capacity} people
            </div>
          </div>
        `)}
      </div>
    </div>
  `
}
```
